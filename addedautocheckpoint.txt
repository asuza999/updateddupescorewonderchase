local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
   Name = "WONDER CHASE DUPE SCRIPT",
   LoadingTitle = "WONDER CHASE DUPE SCRIPT",
   LoadingSubtitle = "BY NYZ SCRIPTS",
   ConfigurationSaving = {
      Enabled = false,
   },
   KeySystem = false,
})

local Tab = Window:CreateTab("MAIN", 4483362458)

local VirtualUser = game:GetService("VirtualUser")
game:GetService("Players").LocalPlayer.Idled:Connect(function()
   VirtualUser:Button2Down(Vector2.new(0,0), workspace.CurrentCamera.CFrame)
   task.wait(1)
   VirtualUser:Button2Up(Vector2.new(0,0), workspace.CurrentCamera.CFrame)
end)

Tab:CreateParagraph({Title = "IMPORTANT NOTE", Content = "BEFORE YOU GO TO THE PORTAL MAKE SURE THE TIMER IS 150S"})

local AutoPickups = false
local CollectDelay = 0.1
local AutoCheckpoint = false

Tab:CreateButton({
   Name = "AUTO CHECKPOINT",
   Callback = function()
      AutoCheckpoint = not AutoCheckpoint
      if AutoCheckpoint then
         task.spawn(function()
            while AutoCheckpoint do
               pcall(function()
                  local folder = workspace:FindFirstChild("ActiveStickers")
                  if folder then
                     local children = folder:GetChildren()
                     if #children == 0 then
                        task.wait(1) 
                     else
                        for _, v in pairs(children) do
                           if not AutoCheckpoint then break end
                           local player = game.Players.LocalPlayer
                           if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                              local targetCFrame = nil
                              if v:IsA("Model") then
                                 targetCFrame = v:GetPivot()
                              elseif v:IsA("BasePart") then
                                 targetCFrame = v.CFrame
                              end
                              if targetCFrame then
                                 player.Character.HumanoidRootPart.CFrame = targetCFrame
                                 task.wait(CollectDelay)
                              end
                           end
                        end
                     end
                  end
               end)
               task.wait(0.1)
            end
         end)
      end
   end,
})

Tab:CreateButton({
   Name = "AUTO COLLECT PICKUPS",
   Callback = function()
      AutoPickups = not AutoPickups
      if AutoPickups then
         task.spawn(function()
            while AutoPickups do
               pcall(function()
                  local player = game.Players.LocalPlayer
                  if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                     local pickups = workspace:FindFirstChild("Pickups")
                     if pickups then
                        for _, v in pairs(pickups:GetChildren()) do
                           if not AutoPickups then break end
                           local targetPart = v:IsA("BasePart") and v or v:FindFirstChildWhichIsA("BasePart", true)
                           if targetPart then
                              player.Character.HumanoidRootPart.CFrame = targetPart.CFrame
                              task.wait(CollectDelay)
                           end
                        end
                     end
                  end
               end)
               task.wait(0.1)
            end
         end)
      end
   end,
})

Tab:CreateButton({
   Name = "TP TO PORTAL",
   Callback = function()
      pcall(function()
         local player = game.Players.LocalPlayer
         if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local portal = workspace.Utility.Teleporters.Portal_to_Wc.Portal
            if portal then
               local targetCFrame = portal:IsA("Model") and portal:GetPivot() or portal.CFrame
               player.Character.HumanoidRootPart.CFrame = targetCFrame
               Rayfield:Notify({Title = "TELEPORT", Content = "TELEPORTED", Duration = 2})
            end
         end
      end)
   end,
})

Tab:CreateButton({
   Name = "DUPE SCORE",
   Callback = function()
      local args = {51, 138.01097051240504}
      local RS = game:GetService("ReplicatedStorage")
      pcall(function()
          RS.Packages.Knit.Services.ObbyService.RF.UpdatePlayerScore:InvokeServer(unpack(args))
      end)
   end,
})

Tab:CreateSlider({
   Name = "CHANGE SPEED",
   Range = {0, 1},
   Increment = 0.01,
   Suffix = "DELAY",
   CurrentValue = 0.1,
   Flag = "CollectSpeed",
   Callback = function(Value)
      CollectDelay = Value
   end,
})

Tab:CreateButton({
   Name = "REJOIN SERVER",
   Callback = function()
      game:GetService("TeleportService"):Teleport(game.PlaceId, game.Players.LocalPlayer)
   end,
})
